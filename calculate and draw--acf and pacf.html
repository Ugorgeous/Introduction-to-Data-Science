def white_noise_series(e_num,e_len):
    '''生成服从标准正太分布的白噪声序列
    e_num为白噪声序列数，e_len为单个序列长度'''
    e=[]
    for i in range(e_num):
        e.append(np.random.randn(e_len))
        
    return e
